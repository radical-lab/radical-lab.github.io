def frames(pictureList,spacing,background=white):
    # Datatype validation checkpoint.
    (borderControl)=(makeEmptyPicture(1,1))
    (increment)=(0)
    while((increment)<(len(pictureList))):
        if((type(pictureList[increment]))!=(type(borderControl))):
            return(showError("UNRECOGNISED PICTURE OBJECT FORMAT!"))
        (increment)+=(1)
    if((type(spacing))!=(int)):
        return(showError("SPACING MUST BE AN INTEGER!"))
    try:
        (pixelObject)=(getPixelAt(borderControl,0,0))
        (setColor(pixelObject,background))
    except:
        return(showError("BACKGROUND MUST BE A COLOUROBJECT!"))
    # Check!
    (index)=(0)
    (widthMax,heightMax)=(0,0)
    while((index)<(len(pictureList))):
        (width,height)=(getWidth(pictureList[index]),getHeight(pictureList[index]))
        (widthMax)+=(width)
        if(((index)>(0))and((index)<(len(pictureList)))):
            (widthMax)+=(spacing)
        if((heightMax)<(height)):
            (heightMax)=(height)
        (index)+=(1)
    (pictureObject)=(makeEmptyPicture(widthMax,heightMax,background))
    (counter)=(0)
    (widthCache)=(0)
    while((counter)<(len(pictureList))):
        if(counter)==(0):
            (spacer)=(0)
        else:
            (spacer)=(spacing)
        (printNow("INSERTING PICTURE OBJECT "+str(counter+1)+"..."))
        (frame)=(pictureList[counter])
        (width,height)=(getWidth(frame),getHeight(frame))
        (frame)=(pictureList[counter])
        (x)=(0)
        while((x)<(width)):
            (y)=(0)
            while((y)<(height)):
                (pixelObject)=(getPixelAt(pictureObject,x+widthCache+spacer,y))
                (setColor(pixelObject,getColor(getPixelAt(frame,x,y))))
                (y)+=(1)
            (x)+=(1)
        (widthCache)+=(width+spacer)
        (counter)+=(1)
    return(pictureObject)

# I just used Bresenham's line algorithm...
def Bresenham(pictureObject,colourObject,x0,y0,x1,y1):
    (width,height)=(getWidth(pictureObject),getHeight(pictureObject))
    # Datatype validation checkpoint.
    (borderControl)=(makeEmptyPicture(1,1))
    if((type(pictureObject))!=(type(borderControl))):
        return(showError("UNRECOGNISED PICTURE OBJECT FORMAT!"))
    if((type(colourObject))!=(type(makeColor(0)))):
        return(showError("UNRECOGNISED COLOUR OBJECT FORMAT!"))
    if(((type(x0))!=(int))or((type(y0))!=(int))or((type(x1))!=(int))or((type(y1))!=(int))):
        return(showError("UNRECOGNISED COORDINATES!"))
    if(((x0)<(0))or((y0)<(0))or((x1)<(0))or((y1)<(0))or((width-1)<(x0))or((height-1)<(y0))or((width-1)<(x1))or((height-1)<(y1))):
        return(showError("OUT OF RANGE COORDINATES!"))
    # Checked!
    (deltaX,deltaY)=(abs(x1-x0),abs(y1-y0))
    if(x0<x1):
        (sX)=(1)
    else:
        (sX)=(-1)
    if(y0<y1):
        (sY)=(1)
    else:
        (sY)=(-1)
    (error)=(deltaX-deltaY)
    while(1):
        (pictureElement)=(getPixelAt(pictureObject,x0,y0))
        (setColor(pictureElement,colourObject))
        if((x0==x1)and(y0==y1)):break
        (err)=(2*error)
        if((err)>(-deltaY)):
            (error)+=(-deltaY)
            (x0)+=(sX)
        if((err)<(deltaX)):
            (error)+=(deltaX)
            (y0)+=(sY)
    return(pictureObject)
