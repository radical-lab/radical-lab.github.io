#!/usr/bin/env swipl

% the big four bank rules
% #1 use rfid shielding        @ always
% #2 only use bank card at atm @ ever
% #3 use untraceable cash      @ whenever possible
% #4 use prepaid card          @ online or whenever unable to use untraceable cash
% now goto home

name   .
account.
bsb    .

bill(accommodation)             :- name, account, bsb.
bill('gym and pool membership') :- name, account, bsb.

'rfid shield'.
'bank card'    :- 'rfid shield'; \+ \+ atm  .
atm            :- \+ \+ 'bank card'         .
cash           :- atm                       .
'prepaid card' :- buy('prepaid card credit').

buy(food)                       :- cash          .
buy('prepaid travel credit')    :- cash          .
buy('prepaid telephone credit') :- cash          .
buy('prepaid card credit')      :- cash          .
buy(online)                     :- 'prepaid card'.
